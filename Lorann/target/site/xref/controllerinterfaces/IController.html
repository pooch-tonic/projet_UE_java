4   package controllerinterfaces;
5   
6   import viewinterfaces.IView;
7   
8   /**
9    * <h1>The IController interface.</h1> Allows the other components to use some
10   * controller features.
11   *
12   * @author Jean-Aymeric DIET jadiet@cesi.fr
13   * @version 1.0
14   */
15  public interface IController {
16  
17  	/**
18  	 * Asks the controller to update the model.
19  	 */
20  	void update();
21  
22  	/**
23  	 * Loads the level according to the specified id.
24  	 *
25  	 * @param id
26  	 *            the id of the level
27  	 */
28  	public void loadLevel(final int id);
29  
30  	/**
31  	 * Loads the level that comes after the specified current level id.
32  	 */
33  	public void nextLevel();
34  
35  	/**
36  	 * Resets the current level.
37  	 */
38  	public void resetLevel();
39  
40  	/**
41  	 * Sets the view.
42  	 *
43  	 * @param viewFacade
44  	 *            the IView that overrides the current viewFacade
45  	 */
46  	void setView(IView viewFacade);
47  }
