4   package model;
5   
6   import enums.UnitTypeEnum;
7   import model.behaviorstrategy.DoNotBounce;
8   import model.behaviorstrategy.DoNotDodge;
9   import model.behaviorstrategy.MoveSimple;
10  import modelinterfaces.IPlayer;
11  import showboard.ISpriteSet;
12  import vector.IVector;
13  import vector.Vector;
14  
15  /**
16   * <h1>The Player class.</h1> The class that represents the player in the game.
17   * 
18   * @author Ryo SHIINA
19   *
20   */
21  public class Player extends Entity implements IPlayer {
22  	public static int SCORE = 0;
23  
24  	/**
25  	 * Instantiates a new Player
26  	 *
27  	 */
28  	public Player(final ISpriteSet spriteSet) {
29  		super(UnitTypeEnum.PLAYER, spriteSet);
30  		this.setLastDirection(new Vector(0, 0));
31  		this.setBounceStrategy(new DoNotBounce());
32  		this.setDodgeStrategy(new DoNotDodge());
33  		this.setMoveStrategy(new MoveSimple());
34  		this.setScoreValue(SCORE);
35  	}
36  
37  	/**
38  	 * Instantiates a new Player
39  	 *
40  	 * @param position
41  	 */
42  	public Player(final IVector position) {
43  		super(position, UnitTypeEnum.PLAYER);
44  		this.setScoreValue(SCORE);
45  	}
46  
47  	@Override
48  	public void update() {
49  		super.update();
50  		this.getSpriteSet().setNextSprite();
51  	}
52  
53  }
