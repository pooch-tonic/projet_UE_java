4   package controller;
5   
6   import java.sql.SQLException;
7   
8   import modelinterfaces.IModel;
9   import showboard.IPawn;
10  import showboard.ISquare;
11  import viewinterfaces.IView;
12  
13  /**
14   * <h1>The LevelLoader class.</h1> Allows to forward the level content to the
15   * view's Showboard.
16   *
17   * @author Ryo
18   *
19   */
20  public class LevelLoader {
21  
22  
23  	/**
24  	 * Loads the level from the MySQL database to put it into the board.
25  	 *
26  	 * @param idLevel
27  	 *            the level id
28  	 * @param model
29  	 *            the model
30  	 * @param view
31  	 *            the view
32  	 */
33  	public void loadLevel(final int idLevel, final IModel model, final IView view) {
34  
35  		try {
36  			model.loadLevel(idLevel);
37  		} catch (final SQLException e) {
38  			e.printStackTrace();
39  		}
40  		view.removeAllPawnsFromBoard();
41  		final ISquare[][] squares = model.getLevel().getUnits();
42  		for (int x = 0; x < squares.length; x++) {
43  			for (int y = 0; y < squares[0].length; y++) {
44  				view.getBoardFrame().addSquare(squares[x][y], x, y);
45  			}
46  		}
47  
48          for (final IPawn entity : model.getLevel().getEntities()) {
49              view.getBoardFrame().addPawn(entity);
50          }
51      }
52  
53  
54  	/**
55  	 * Generates the next level.
56  	 *
57  	 * @param model
58  	 *            the model
59  	 * @param view
60  	 *            the view
61  	 */
62  	public void loadNextLevel(final IModel model, final IView view) {
63  		this.loadLevel(model.getLevel().getId() + 1, model, view);
64  
65      }
66  
67  	/**
68  	 * Regenerates the current level.
69  	 *
70  	 * @param model
71  	 *            the model
72  	 * @param view
73  	 *            the view
74  	 */
75  	public void resetLevel(final IModel model, final IView view) {
76  		this.loadLevel(model.getLevel().getId(), model, view);
77  	}
78  
79  }
